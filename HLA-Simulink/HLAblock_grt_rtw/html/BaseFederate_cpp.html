56//Connect to RTI. In order to connect we need to create federation based on FOM (isn't nessary, if federation already exist)



57// and join there. After that federate initialized in RTI and go to the his main loop.



58//Use only to lvalue class samples



59    bool BaseFederate::ConnectRTI() {



60    //Initialized rtiAmbassador to call RTI servecies (look HLA::MakeRTIambassador)



61        Logger log(_log_filename);



62        try{



63            _rtiAmbassador = MakeRTIambassador();



64        }



65        catch(RTIinternalError){



66            log << L"ERROR:" << _federate_name << L"Can't create RTIambassador" << Logger::Flush();



67            return false;



68        }



69



70        try{



71    //Check the CRC address and callback RTI mode (sync, async) and using rtiAmbassador connect this federate to RTI



72            if(_host_IP_address == L"localhost")



73                _rtiAmbassador->connect(*this,_callback_mode);



74            else



75                _rtiAmbassador->connect(*this,_callback_mode,L"crcAddress="+_host_IP_address);



76        }



77        catch(...){



78            log << L"ERROR:" << _federate_name << L"Cant connect" << Logger::Flush();



79            return false;



80        }



81



82



83        try{



84    //Try to create federation with name (_federation_name) and rules in FOM (look _FOMname)



85            _rtiAmbassador->createFederationExecution(_federation_name, _FOMname);



86        }



87    //If federation already exist we catch exception(FederationExecutionAlreadyExists) and do nothing



88        catch(FederationExecutionAlreadyExists&){}



89



90        catch(ErrorReadingFDD&){



91            log << L"ERROR:" << _federate_name << L"Cant read FOM" << Logger::Flush();



92            return false;



93        }



94        catch(CouldNotOpenFDD&){



95            log << L"ERROR:" << _federate_name << L"Cant open FOM" << Logger::Flush();



96            return false;



97        }



98



99        catch(...){



100            log << L"ERROR:" << _federate_name << L"Cant create federation" << Logger::Flush();



101            return false;



102        }



103



104    //This waste 1 seconds, it's the biggest comand in this function (more than 50ths)



105        try{



106    //After connect and create/find federation with name (_federation_name) we join to it with name (_federate_name)



107            _rtiAmbassador->joinFederationExecution(_federate_name,_federation_name);



108            log << L"INFO:" << L"Connect of" << _federate_name <<  L"done" << Logger::Flush();



109        }



110        catch(...){



111            log << L"ERROR:" << _federate_name << L"Can't join" << Logger::Flush();



112            return false;



113        }



114



115    //If all connect steps finished we set connect flag to true value



116        _state = State::CONNECTED;



117        try{



118    //Initialized federate (it's object type in FOM and attributes), environment in federation



119    // (other types and attributes indicated in _ObjectsNames) and their connections for this federate



120            Init();



121



122            log << L"INFO:" << L"Init of" << _federate_name << L"done" << Logger::Flush();



123        }



124        catch(RTIinternalError& e){



125            log << L"ERROR:" << _federate_name << L"Error in Init() with" << e.what() << Logger::Flush();



126            return false;



127        }



128



129        try{



130    //Run the main loop of federate



131            _f_modeling = true;



132            if(_mode == ModelMode::THREADING)



133                _modeling_thread = std::thread(&BaseFederate::ThreadModeling,this);



134            else



135                _last_time = chrono::steady_clock::now();



136            RunFederate();



137        }



138        catch(RTIinternalError& e){



139            log << L"ERROR:" << _federate_name << L"Error in Run() with" << e.what() << Logger::Flush();



140            return false;



141        }



142        return true;



143    }

152    void BaseFederate::Init(){



153



154        unordered_map<ObjectClassHandle,AttributeHandleSet,ObjectClassHash> _externAttributesSet;



155



156        AttributeHandleSet _aPublishSet;



157    //Initializerd federate object and his attributes and environmental objects and attributes indicated in FOM

159

161        SubscribeAttributes(_externAttributesSet);

164        PublishAttributes(_aPublishSet);

170



171        PublishParameters();



172

174



175        _state = State::STARTED;

178//Initializerd federate object and his attributes and environmental objects and attributes indicated in _ObjectNames



179    void BaseFederate::InitClassesAndAttributes(unordered_map<ObjectClassHandle,AttributeHandleSet,ObjectClassHash>& _externAttributesSet,



180                                                AttributeHandleSet& _aPublishSet){



181



182    //Initializerd our federate object(_MyClass) from FOM refer to _federate_type



183        _MyClass = _rtiAmbassador->getObjectClassHandle(_federate_type);



184



185        for(const auto& Attribute:_AttributeNames){



186        //Set table(_AttributesMap) of this object attributes like [HLAobjectClass(something like link to FOM), [AttributeName, HLAattribute(something like a link to FOM]]



187            _AttributesMap[_MyClass][Attribute] = _rtiAmbassador->getAttributeHandle(_MyClass,Attribute);



188



189        //Insert federate attributes to Publish Set(_aPublishSetId)



190            _aPublishSet.insert(_AttributesMap[_MyClass][Attribute]);



191        }



192



193    //Initializerd environmental objects and attributes indicated in _ObjectNames and attributes of it federate



194        for(const auto&Object:_ObjectsNames){



195



196        //Cache of Object Name



197            auto& objectName = Object.first;



198



199        //Set table(_ObjectClasses) of objects like [TypeName, HLAobjectClass(something like link to FOM)]



200            _ObjectClasses[objectName] = _rtiAmbassador->getObjectClassHandle(objectName);



201



202            for(const auto&Attribute:Object.second){



203            //Cache of Object ID



204                auto& objectId = _ObjectClasses[objectName];



205



206            //Set table(_AttributesMap) of object's attributes like [HLAobjectClass(something like link to FOM), [AttributeName, HLAattribute(something like a link to FOM]]



207                _AttributesMap[objectId][Attribute] =_rtiAmbassador->getAttributeHandle(objectId,Attribute);



208



209            //Insert extern attributes to Subscribe Set(_externAttributeSet)



210                _externAttributesSet[objectId].insert(_AttributesMap[objectId][Attribute]);



211



212            }



213        }



214    }

217    void BaseFederate::InitInteractionsAndParameters(){



218



219        for(const auto&Interaction:_InteractionsNames){



220        //Cache of Interaction Name



221            auto& interactionName = Interaction.first;



222



223        //Set table(_InteractionClasses) of interactions like [InteractionTypeName, HLAInteractionClass(something like link to FOM)]



224            _InteractionClasses[interactionName] = _rtiAmbassador->getInteractionClassHandle(interactionName);



225



226            for(const auto&Parameter:Interaction.second){



227            //Cache of InteractionId



228                auto& interactionId = _InteractionClasses[Interaction.first];



229



230            //Set table(_ParametersMap) of interaction's parameters like [HLAInteractionClass(something like link to FOM), [ParametersName, HLAparameter(something like a link to FOM]]



231                _ParametersMap[interactionId][Parameter] =_rtiAmbassador->getParameterHandle(interactionId,Parameter);



232            }



233        }



234    }

285    void BaseFederate::ThreadModeling(){



286        while(_f_modeling){



287            Logger log(_log_filename);



288            {



289                lock_guard<mutex> guard(_smutex);



290                _state = State::DOING;



291                log << L"INFO:" << _federate_name << L"Begin of threading modeling step - doing" << Logger::Flush();



292            }



293            _cond.notify_one();



294            this_thread::sleep_for(chrono::milliseconds(_modeling_step));



295            log << L"INFO:" << _federate_name << L"End of threading modeling step - proccessing" << Logger::Flush();



296            UpdateAttributes();



297            thread ParametersThread(&BaseFederate::ParameterProcess,this);



298            ParametersThread.detach();



299            AttributeProcess();



300        }



301    }

303    void BaseFederate::FollowModeling(){



304        Logger log(_log_filename);



305        auto dur = chrono::duration_cast<chrono::milliseconds>(chrono::steady_clock::now()-_last_time);



306        auto step = chrono::duration_cast<chrono::milliseconds>(chrono::milliseconds(_modeling_step));



307        if((step - dur).count()>0)



308            this_thread::sleep_for(chrono::milliseconds(step - dur));



309        log << L"INFO:" << _federate_name << L"End of following modeling step - proccessing" << Logger::Flush();



310        UpdateAttributes();



311        ParameterProcess();



312        AttributeProcess();



313        _state = State::DOING;



314        log << L"INFO:" << _federate_name << L"Begin of following modeling step - doing" << Logger::Flush();



315    }
