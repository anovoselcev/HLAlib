55//Connect to RTI. In order to connect we need to create federation based on FOM (isn't nessary, if federation already exist)



56// and join there. After that federate initialized in RTI and go to the his main loop.



57//Use only to lvalue class samples



58    void UPIMFederate::ConnectRTI() &{



59    //Initialized rtiAmbassador to call RTI servecies (look UPIM::MakeRTIambassador)



60        _rtiAmbassador = MakeRTIambassador();



61        try{



62



63    //Check the CRC address and callback RTI mode (sync, async) and using rtiAmbassador connect this federate to RTI



64            if(_host_IP_address == L"localhost")



65                _rtiAmbassador->connect(*this,_callback_mode);



66            else



67                _rtiAmbassador->connect(*this,_callback_mode,L"crcAddress="+_host_IP_address);



68        }



69



70        catch(exception&){wcout << L"Cant connect" << endl; return;}



71



72		try{



73    //Try to create federation with name (_federation_name) and rules in FOM (look _FOMname)



74            _rtiAmbassador->createFederationExecution(_federation_name, _FOMname);



75		}



76



77    //If federation already exist we catch exception(FederationExecutionAlreadyExists) and do nothing



78		catch(FederationExecutionAlreadyExists&){}



79



80        try{



81    //After connect and create/find federation with name (_federation_name) we join to it with name (_federate_name)



82            _rtiAmbassador->joinFederationExecution(_federate_name,_federation_name);



83            wcout << L"Connect done" << endl;



84        }



85        catch(Exception&){wcout << L"Can't join" << endl; return;}



86



87    //If all connect steps finished we set connect flag to true value



88		_f_connect = true;



89		try{



90    //Initialized federate (it's object type in FOM and attributes), environment in federation



91    // (other types and attributes indicated in _ObjectsNames) and their connections for this federate



92			Init();



93



94            wcout << L"Init done" << endl;



95		}



96		catch(RTIinternalError& e){



97            wcout << L"Error in Init() with " << endl;



98        }



99		try{



100



101    //Run the main loop of federate



102            _f_modeling = true;



103            _modeling_thread = std::thread(&UPIMFederate::Modeling,this);



104            RunFederate();



105		}



106		catch(RTIinternalError& e){



107            wcout << L"Error in Run() with " << e.what() << endl;



108		}



109	}

143//Initializerd federate object and his attributes and environmental objects and attributes indicated in _ObjectNames



144	void UPIMFederate::InitClassesAndAttributes(){



145



146    //Initializerd our federate object(_MyClass) from FOM refer to _federate_type



147		_MyClass = _rtiAmbassador->getObjectClassHandle(_federate_type);



148



149        for(const auto& Attribute:_AttributeNames){



150        //Set table(_AttributesMap) of this object attributes like [HLAobjectClass(something like link to FOM), [AttributeName, HLAattribute(something like a link to FOM]]



151            _AttributesMap[_MyClass][Attribute] = _rtiAmbassador->getAttributeHandle(_MyClass,Attribute);



152



153        //Insert federate attributes to Publish Set(_aPublishSetId)



154            _aPublishSet.insert(_AttributesMap[_MyClass][Attribute]);



155        }



156



157    //Initializerd environmental objects and attributes indicated in _ObjectNames and attributes of it federate



158        for(const auto&Object:_ObjectsNames){



159



160        //Cache of Object Name



161            auto& objectName = Object.first;



162



163        //Set table(_ObjectClasses) of objects like [TypeName, HLAobjectClass(something like link to FOM)]



164            _ObjectClasses[objectName] = _rtiAmbassador->getObjectClassHandle(objectName);



165



166            for(const auto&Attribute:Object.second){



167            //Cache of Object ID



168                auto& objectId = _ObjectClasses[objectName];



169



170            //Set table(_AttributesMap) of object's attributes like [HLAobjectClass(something like link to FOM), [AttributeName, HLAattribute(something like a link to FOM]]



171                _AttributesMap[objectId][Attribute] =_rtiAmbassador->getAttributeHandle(objectId,Attribute);



172



173            //Insert extern attributes to Subscribe Set(_externAttributeSet)



174                _externAttributesSet[objectId].insert(_AttributesMap[objectId][Attribute]);



175



176			}



177		}



178	}

180//Initializerd federate interactions and his parameters and environmental interactions and parameters indicated in _InteractionNames



181	void UPIMFederate::InitInteractionsAndParameters(){



182



183        for(const auto&Interaction:_InteractionsNames){



184        //Cache of Interaction Name



185            auto& interactionName = Interaction.first;



186



187        //Set table(_InteractionClasses) of interactions like [InteractionTypeName, HLAInteractionClass(something like link to FOM)]



188            _InteractionClasses[interactionName] = _rtiAmbassador->getInteractionClassHandle(interactionName);



189



190            for(const auto&Parameter:Interaction.second){



191            //Cache of InteractionId



192                auto& interactionId = _InteractionClasses[Interaction.first];



193



194            //Set table(_ParametersMap) of interaction's parameters like [HLAInteractionClass(something like link to FOM), [ParametersName, HLAparameter(something like a link to FOM]]



195                _ParametersMap[interactionId][Parameter] =_rtiAmbassador->getParameterHandle(interactionId,Parameter);



196			}



197		}



198	}

233	void UPIMFederate::RegisterName(){



234        _rtiAmbassador->reserveObjectInstanceName(_federate_name);



235        auto start = chrono::steady_clock::now();



236        auto dur = chrono::duration_cast<chrono::microseconds>(chrono::steady_clock::now()-start).count();



237        while(!_f_name_reserve || dur<=3000){



238            dur = chrono::duration_cast<std::chrono::microseconds>(chrono::steady_clock::now()-start).count();



239        }



240          _MyInstanceID = _rtiAmbassador->registerObjectInstance(_MyClass, _federate_name);



241	}

255    void UPIMFederate::Modeling(){



256        while(_f_modeling){



257            _state = State::PREPARE;



258            std::this_thread::sleep_for(std::chrono::milliseconds(_modeling_step));



259            _state = State::READY;



260            AttributeProcess();



261            ParameterProcess();



262            UpdateAttributes();



263        }



264    }
