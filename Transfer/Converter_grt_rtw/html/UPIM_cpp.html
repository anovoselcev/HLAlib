51//Connect to RTI. In order to connect we need to create federation based on FOM (isn't nessary, if federation already exist)



52// and join there. After that federate initialized in RTI and go to the his main loop.



53//Use only to lvalue class samples



54	void UPIMFederate::ConnectRRTI() &{



55    //Vector that keep path to FOM



56		vector<wstring> FOMmoduleUrls;



57		FOMmoduleUrls.push_back(_FOMname);



58



59    //Initialized rtiAmbassador to call RTI servecies (look UPIM::MakeRTIambassador)



60        _rtiAmbassador = MakeRTIambassador();



61        try{



62



63    //Check the CRC address and callback RTI mode (sync, async) and using rtiAmbassador connect this federate to RTI



64        if(_host_IP_address == L"localhost")



65            _rtiAmbassador->connect(*this,_callback_mode);



66        else



67            _rtiAmbassador->connect(*this,_callback_mode,L"crcAddress="+_host_IP_address);



68        }



69        catch(exception&){wcout<<L"Cant connect"<<endl; return;}



70        {



71		try{



72



73    //Try to create federation with name (_federation_name) and rules in FOM (look _FOMname)



74			_rtiAmbassador->createFederationExecution(_federation_name, FOMmoduleUrls);



75		}



76



77    //If federation already exist we catch exception(FederationExecutionAlreadyExists) and do nothing



78		catch(FederationExecutionAlreadyExists&){}



79



80    //After connect and create/find federation with name (_federation_name) we join to it with name (_federate_name)



81		_rtiAmbassador->joinFederationExecution(_federate_name,_federation_name);



82        }



83        wcout<<L"Connect done"<<endl;



84



85    //If all connect steps finished we set connect flag to true value



86		_f_connect = true;



87		try{



88    //Initialized federate (it's object type in FOM and attributes), environment in federation



89    // (other types and attributes indicated in _ObjectsNames) and their connections for this federate



90			Init();



91



92            wcout<<L"Init done"<<endl;



93		}



94		catch(RTIinternalError& e){



95            wcout<<L"Error in Init() with "<<endl;



96        }



97		try{



98



99    //Run the main loop of federate



100			RunFederate();



101		}



102		catch(RTIinternalError& e){



103            wcout<<L"Error in Run() with "<<e.what()<<endl;



104		}



105	}

134//Initializerd federate object and his attributes and environmental objects and attributes indicated in _ObjectNames



135	void UPIMFederate::InitClassesAndAttributes(){



136



137    //Initializerd our federate object(_MyClass) from FOM refer to _federate_type



138		_MyClass = _rtiAmbassador->getObjectClassHandle(_federate_type);



139



140        for(const auto& Attribute:_AttributeNames){



141        //Set table(_AttributesMap) of this object attributes like [HLAobjectClass(something like link to FOM), [AttributeName, HLAattribute(something like a link to FOM]]



142            _AttributesMap[_MyClass][Attribute] = _rtiAmbassador->getAttributeHandle(_MyClass,Attribute);



143



144        //Insert federate attributes to Publish Set(_aPublishSetId)



145            _aPublishSetId.insert(_AttributesMap[_MyClass][Attribute]);



146        }



147



148    //Initializerd environmental objects and attributes indicated in _ObjectNames and attributes of it federate



149        for(const auto&Object:_ObjectsNames){



150



151        //Cache of Object Name



152            auto& objectName = Object.first;



153



154        //Set table(_ObjectClasses) of objects like [TypeName, HLAobjectClass(something like link to FOM)]



155            _ObjectClasses[objectName] = _rtiAmbassador->getObjectClassHandle(objectName);



156



157            for(const auto&Attribute:Object.second){



158            //Cache of Object ID



159                auto& objectId = _ObjectClasses[objectName];



160



161            //Set table(_AttributesMap) of object's attributes like [HLAobjectClass(something like link to FOM), [AttributeName, HLAattribute(something like a link to FOM]]



162                _AttributesMap[objectId][Attribute] =_rtiAmbassador->getAttributeHandle(objectId,Attribute);



163



164            //Insert extern attributes to Subscribe Set(_externAttributeSet)



165                _externAttributesSet[objectId].insert(_AttributesMap[objectId][Attribute]);



166



167			}



168		}



169	}

171//Initializerd federate interactions and his parameters and environmental interactions and parameters indicated in _InteractionNames



172	void UPIMFederate::InitInteractionsAndParameters(){



173



174        for(const auto&Interaction:_InteractionsNames){



175        //Cache of Interaction Name



176            auto& interactionName = Interaction.first;



177



178        //Set table(_InteractionClasses) of interactions like [InteractionTypeName, HLAInteractionClass(something like link to FOM)]



179            _InteractionClasses[interactionName] = _rtiAmbassador->getInteractionClassHandle(interactionName);



180



181            for(const auto&Parameter:Interaction.second){



182            //Cache of InteractionId



183                auto& interactionId = _InteractionClasses[Interaction.first];



184



185            //Set table(_ParametersMap) of interaction's parameters like [HLAInteractionClass(something like link to FOM), [ParametersName, HLAparameter(something like a link to FOM]]



186                _ParametersMap[interactionId][Parameter] =_rtiAmbassador->getParameterHandle(interactionId,Parameter);



187			}



188		}



189	}

194



195    //Send to RTI request to subscribe on Object(x.first) and his Attribute Set (x.second)

197			_rtiAmbassador->subscribeObjectClassAttributes(x.first,x.second);

224	void UPIMFederate::RegisterName(){



225        _rtiAmbassador->reserveObjectInstanceName(_federate_name);



226        auto start = chrono::steady_clock::now();



227        auto dur = chrono::duration_cast<chrono::microseconds>(chrono::steady_clock::now()-start).count();



228        while(!_f_name_reserve || dur<=3000){



229            dur = chrono::duration_cast<std::chrono::microseconds>(chrono::steady_clock::now()-start).count();



230        }



231          _MyInstanceID = _rtiAmbassador->registerObjectInstance(_MyClass, _federate_name);



232	}
