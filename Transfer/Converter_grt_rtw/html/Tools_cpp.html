199    timespec timespecAdd(timespec t1, timespec t2) {



200      constexpr long BILLION = 1000000000;



201      timespec ret;



202      ret.tv_sec = t2.tv_sec + t1.tv_sec;



203      ret.tv_nsec = t2.tv_nsec + t1.tv_nsec;



204      if (ret.tv_nsec >= BILLION) {



205        ret.tv_nsec -= BILLION;



206        ret.tv_sec++;



207      }



208      return ret;



209    }

294    std::string UTF16toUTF8(const wstring & in)



295    {



296      string out;



297      unsigned int codepoint=0;



298      bool completecode = false;



299      wstring::const_iterator p;



300      for (p = in.begin();  p != in.end();  ++p)



301      {



302        if (*p >= 0xd800 && *p <= 0xdbff)



303        {



304          codepoint = ((*p - 0xd800) << 10) + 0x10000;



305          completecode = false;



306        }



307        else if (!completecode && *p >= 0xdc00 && *p <= 0xdfff)



308        {



309          codepoint |= *p - 0xdc00;



310          completecode = true;



311        }



312        else



313        {



314          codepoint = *p;



315          completecode = true;



316        }



317        if (completecode)



318        {



319          if (codepoint <= 0x7f)



320            out.push_back(codepoint);



321          else if (codepoint <= 0x7ff)



322          {



323            out.push_back(0xc0 | ((codepoint >> 6) & 0x1f));



324            out.push_back(0x80 | (codepoint & 0x3f));



325          }



326          else if (codepoint <= 0xffff)



327          {



328            out.push_back(0xe0 | ((codepoint >> 12) & 0x0f));



329            out.push_back(0x80 | ((codepoint >> 6) & 0x3f));



330            out.push_back(0x80 | (codepoint & 0x3f));



331          }



332          else



333          {



334            out.push_back(0xf0 | ((codepoint >> 18) & 0x07));



335            out.push_back(0x80 | ((codepoint >> 12) & 0x3f));



336            out.push_back(0x80 | ((codepoint >> 6) & 0x3f));



337            out.push_back(0x80 | (codepoint & 0x3f));



338          }



339        }



340      }



341      return out;



342    }

354    string getTimeSinceEpoch()



355    {



356      char mbstr[100];



357      char buf[4];



358



359      std::chrono::time_point<std::chrono::system_clock> t = std::chrono::system_clock::now();



360      std::time_t tm_t = std::chrono::system_clock::to_time_t(t);



361      struct tm result_tm;



362#ifdef WIN32



363      gmtime_s(&result_tm, &tm_t);



364#else



365      gmtime_r(&tm_t, &result_tm);



366#endif



367      auto t_msec= std::chrono::duration_cast<std::chrono::milliseconds>(t.time_since_epoch()).count() %1000;



368      std::strftime(mbstr, 100, "%y.%m.%d %H:%M:%S.", &result_tm);



369#ifdef WIN32



370      sprintf_s(buf,4,"%3.3d",(int)t_msec);



371#else



372      snprintf(buf,4,"%3.3d",(int)t_msec);



373#endif



374      return std::string(mbstr) + buf;



375    }
