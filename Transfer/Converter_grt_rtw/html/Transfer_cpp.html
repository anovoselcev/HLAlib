75    void RTIStaff::getDataMax(void* ptrSource, unsigned long uiMaxSize){



76        HLA::RTIbyte  field0;



77        HLA::RTIfloat64BE field1;



78        HLA::RTIinteger16BE  field2;



79        HLA::RTIstring field3;



80        HLA::RTIchar field4;



81        unsigned offset = 0;



82        // Определяем размер поля данных



83        F_offset< HLA::RTIbyte, HLA::RTIfloat64BE >(field0,field1,offset,ptrSource,uiMaxSize);



84        F_offset< HLA::RTIfloat64BE, HLA::RTIinteger16BE >(field1,field2,offset,ptrSource,uiMaxSize);



85        F_offset< HLA::RTIinteger16BE, HLA::RTIstring >(field2,field3,offset,ptrSource,uiMaxSize);



86        F_offset< HLA::RTIstring, HLA::RTIchar >(field3,field4,offset,ptrSource,uiMaxSize);



87        F_offsetLast< HLA::RTIchar >(field4,offset,ptrSource,static_cast<unsigned>(uiMaxSize));



88



89        // Создаем новое поле данных и копируем туда данные;



90



91        if (ptrData!=nullptr) delete[] ptrData;



92        m_uiSizeData = offset;



93        ptrData = new HLA::Octet[m_uiSizeData];



94        memcpy(ptrData, ptrSource, m_uiSizeData);



95    }

97    void RTIStaff::get(Staff const &obj){



98        HLA::RTIbyte  field0;



99        HLA::RTIfloat64BE field1;



100        HLA::RTIinteger16BE  field2;



101        HLA::RTIstring field3;



102        HLA::RTIchar field4;



103        unsigned offset = 0, P, uiSize, mmOBV;



104        m_uiSizeData=0;



105        // 0-ое поле



106        field0.get(obj.b);



107        uiSize = field0.getsize();



108



109        // 1-ое поле



110        field1.get(obj.f);



111        mmOBV = field1.getOctetBoundary();



112        P = HLA::Tools::getPendingBytes(offset+uiSize,mmOBV);



113        offset += uiSize+P;



114        uiSize = field1.getsize();



115        // 2-ое поле



116        field2.get(obj.i);



117        mmOBV = field2.getOctetBoundary();



118        P = HLA::Tools::getPendingBytes(offset+uiSize,mmOBV);



119        offset += uiSize+P;



120        uiSize = field2.getsize();



121        // 3-e поле



122        field3.get(obj.s);



123        mmOBV = field3.getOctetBoundary();



124        P = HLA::Tools::getPendingBytes(offset+uiSize,mmOBV);



125        offset += uiSize+P;



126        uiSize = field3.getsize();



127        // 4-ое поле



128        field4.get(obj.c);



129        mmOBV = field3.getOctetBoundary();



130        P = HLA::Tools::getPendingBytes(offset+uiSize,mmOBV);



131        offset += uiSize+P;



132        uiSize = field4.getsize();



133



134        offset += uiSize;



135        m_uiSizeData = offset;



136        if (ptrData!=nullptr) delete[] ptrData;



137        ptrData = new HLA::Octet[m_uiSizeData];



138        offset = 0;



139        // 0-ое поле



140        uiSize = field0.getsize();



141        field0.setData(ptrData+offset,uiSize);



142        // 1-ое поле



143        mmOBV = field1.getOctetBoundary();



144        P = HLA::Tools::getPendingBytes(offset+uiSize,mmOBV);



145        offset += uiSize+P;



146        uiSize = field1.getsize();

148        // 2-ое поле
