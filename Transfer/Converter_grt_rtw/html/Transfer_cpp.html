73    void RTIStaff::getDataMax(void* ptrSource, unsigned long uiMaxSize){



74        HLA::RTIbyte  field0;



75        HLA::RTIfloat64BE field1;



76        HLA::RTIinteger16BE  field2;



77        HLA::RTIstring field3;



78        HLA::RTIchar field4;



79        unsigned offset = 0;



80        // Определяем размер поля данных



81        F_offset< HLA::RTIbyte, HLA::RTIfloat64BE >(field0,field1,offset,ptrSource,uiMaxSize);



82        F_offset< HLA::RTIfloat64BE, HLA::RTIinteger16BE >(field1,field2,offset,ptrSource,uiMaxSize);



83        F_offset< HLA::RTIinteger16BE, HLA::RTIstring >(field2,field3,offset,ptrSource,uiMaxSize);



84        F_offset< HLA::RTIstring, HLA::RTIchar >(field3,field4,offset,ptrSource,uiMaxSize);



85        F_offsetLast< HLA::RTIchar >(field4,offset,ptrSource,static_cast<unsigned>(uiMaxSize));



86



87        // Создаем новое поле данных и копируем туда данные;



88



89        if (ptrData!=nullptr) delete[] ptrData;



90        m_uiSizeData = offset;



91        ptrData = new HLA::Octet[m_uiSizeData];



92        memcpy(ptrData, ptrSource, m_uiSizeData);

95    void RTIStaff::get(Staff const &obj){



96        HLA::RTIbyte  field0;



97        HLA::RTIfloat64BE field1;



98        HLA::RTIinteger16BE  field2;



99        HLA::RTIstring field3;



100        HLA::RTIchar field4;



101        unsigned offset = 0, P, uiSize, mmOBV;



102        m_uiSizeData=0;



103        // 0-ое поле



104        field0.get(obj.b);



105        uiSize = field0.getsize();



106



107        // 1-ое поле



108        field1.get(obj.f);



109        mmOBV = field1.getOctetBoundary();



110        P = HLA::Tools::getPendingBytes(offset+uiSize,mmOBV);



111        offset += uiSize+P;



112        uiSize = field1.getsize();



113        // 2-ое поле



114        field2.get(obj.i);



115        mmOBV = field2.getOctetBoundary();



116        P = HLA::Tools::getPendingBytes(offset+uiSize,mmOBV);



117        offset += uiSize+P;



118        uiSize = field2.getsize();



119        // 3-e поле



120        field3.get(obj.s);



121        mmOBV = field3.getOctetBoundary();



122        P = HLA::Tools::getPendingBytes(offset+uiSize,mmOBV);



123        offset += uiSize+P;

126        field4.get(obj.c);



127        mmOBV = field3.getOctetBoundary();



128        P = HLA::Tools::getPendingBytes(offset+uiSize,mmOBV);



129        offset += uiSize+P;



130        uiSize = field4.getsize();



131



132        offset += uiSize;

136        offset = 0;



137        // 0-ое поле

140        // 1-ое поле
